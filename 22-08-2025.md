<b>1)take string as input and convert it into the othercase  </b>

<b>code:</b>

def case(s):

    res=''
    for i in s:
        if 'a'<=i<='z':
            res+=chr(ord(i)-32)
        elif 'A'<=i<='Z':
            res+=chr(ord(i)+32)
        else:
            res+=i
    return res
    
s=input("enter a string: ")

print(case(s))
