<b>5)program to find whether the given is prime or not.</b>

<b>code:</b>

def prime_or_not(n):

    if n<=1:
        return False
    for i in range(2,n//2+1):
        if n%i==0:
            return False
    return "True"
    
n=int(input("enter a number= "))

if prime_or_not(n):

    print(f"{n} is a Prime number.")

else:

    print(f"{n} is Not a Prime number.")
